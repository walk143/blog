# 初始化当前目录为一个仓库
git init
# 添加远程仓库地址
git remote add origin git@gitee.com:sloera/weasel.git
# 拉取远程仓库
git fetch
# 备份当前目录
mv -r * ../tmp
# 从远程分支创建一个本地分支
git checkout -b master origin/master
# 恢复之前文件
mv -r ../tmp .
# 查看状态
git status
# 提交或推送
git commit -am "temp"
git push
